## Order Creation

### createDebridgeBridgeOrder

Create a bridge order for cross-chain token transfer.

```typescript
createDebridgeBridgeOrder(params: deBridgeOrderInput): Promise<deBridgeOrderResponse>
```

<ParamField path="params" type="deBridgeOrderInput" required>
  Bridge order parameters object
  
  <ParamField path="srcChainId" type="string" required>
    Source chain ID
  </ParamField>
  
  <ParamField path="srcChainTokenIn" type="string" required>
    Token address on source chain
  </ParamField>
  
  <ParamField path="srcChainTokenInAmount" type="string" required>
    Amount to bridge in token's smallest unit
  </ParamField>
  
  <ParamField path="dstChainId" type="string" required>
    Destination chain ID (must differ from source)
  </ParamField>
  
  <ParamField path="dstChainTokenOut" type="string" required>
    Token address on destination chain
  </ParamField>
  
  <ParamField path="dstChainTokenOutRecipient" type="string" required>
    Recipient address on destination chain
  </ParamField>
  
  <ParamField path="account" type="string" required>
    Sender's wallet address
  </ParamField>
  
  <ParamField path="dstChainTokenOutAmount" type="string">
    Expected output amount (default: "auto")
  </ParamField>
  
  <ParamField path="referralCode" type="number">
    Referral code for the transaction
  </ParamField>
  
  <ParamField path="affiliateFeePercent" type="number">
    Affiliate fee percentage (must be used with affiliateFeeRecipient)
  </ParamField>
  
  <ParamField path="affiliateFeeRecipient" type="string">
    Address to receive affiliate fees
  </ParamField>
</ParamField>

<ResponseField name="return" type="Promise<deBridgeOrderResponse>">
  Returns bridge order details including transaction data
</ResponseField>

**Location:** `src/utils/deBridge/createDeBridgeOrder.ts:17`

**Example:**
```typescript
const order = await createDebridgeBridgeOrder({
  srcChainId: "1",
  srcChainTokenIn: "0x...",
  srcChainTokenInAmount: "1000000",
  dstChainId: "137",
  dstChainTokenOut: "0x...",
  dstChainTokenOutRecipient: "0x...",
  account: "0x..."
});
```

</ParamField>

<ParamField path="txData" type="string" required>

</ParamField>

<ParamField path="solWallet" type="Keypair" required>
  Signer keypair
</ParamField>

<ResponseField name="return" type="Promise<VersionedTransaction>">
  A signed Solana transaction ready for submission
</ResponseField>

**Location:** `src/utils/solana.ts:12`
